configfile $SERVER_ROOT/conf/vhosts/$VH_NAME/vhconf.conf
enablescript 1
setuidmode 2
restrained 1
allowsymbollink 1
chrootmode 0
vhroot $SERVER_ROOT/$VH_NAME/

virtualhostconfig  {
    docroot $VH_ROOT/public/
    enablegzip 1

    accesscontrol  {
        allow *
    }

    rewrite  {
        enable 0
        loglevel 0
    }

    hotlinkctrl  {
        enablehotlinkctrl 0
        suffixes gif, jpeg, jpg
        allowdirectaccess 1
        onlyself 1
    }

    expires  {
        enableexpires 1
    }

    index  {
        useserver 0
        autoindex 0
        indexfiles index.html
        autoindexuri /_autoindex/default.php
    }

    context / {
        accesscontrol 
        adddefaultcharset off
        railsenv 1
        maxconns 5
        location $VH_ROOT/
        type rails

        rewrite  {
            rules rewritefile .htaccess
            unknownkeywords rewritefile .htaccess
        }
    }

    errorlog  {
        useserver 1
    }

    accesslog $SERVER_ROOT/logs/$VH_NAME.access.log {
        keepdays 30
        rollingsize 500M
        logheaders 3
        compressarchive 0
        useserver 0
    }

    awstats  {
        workingdir $VH_ROOT/awstats
        awstatsuri /awstats/
        sitedomain localhost
        sitealiases 127.0.0.1 localhost
        updatemode 0
        updateinterval 86400
        updateoffset 0
        securedconn 0
    }
}
