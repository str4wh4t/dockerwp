    enablegzip 1
    docroot $VH_ROOT/public/

    accesslog $SERVER_ROOT/logs/$VH_NAME.access.log {
        logheaders 3
        compressarchive 0
        rollingsize 500M
        useserver 0
        keepdays 30
    }

    accesscontrol  {
        allow *
    }

    rewrite  {
        enable 0
        loglevel 0
    }

    hotlinkctrl  {
        enablehotlinkctrl 0
        allowdirectaccess 1
        suffixes gif, jpeg, jpg
        onlyself 1
    }

    errorlog  {
        useserver 1
    }

    context / {
        railsenv 1
        accesscontrol 
        adddefaultcharset off
        maxconns 5

        type rails

        rewrite  {
            rules rewritefile .htaccess
            unknownkeywords rewritefile .htaccess
        }
    }

    index  {
        indexfiles index.html

        autoindex 0

    expires  {

    awstats  {
        workingdir $VH_ROOT/awstats

        sitealiases 127.0.0.1 localhost
        updatemode 0
        updateinterval 86400
        updateoffset 0
        securedconn 0
